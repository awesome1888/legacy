export const demolitionIndexQuery = graphql`
    query demolitionIndexQuery {
        allMdx(
            sort: { fields: [frontmatter___shortTitle], order: ASC }
            filter: {
                frontmatter: {
                    published: { eq: true }
                    status: { eq: "demolition" }
                }
            }
        ) {
            edges {
                node {
                    id
                    frontmatter {
                        title
                        shortTitle
                        slug
                        description
                        published
                        headerImage
                        images {
                            image {
                                childImageSharp {
                                    fluid(maxWidth: 1240, quality: 80) {
                                        ...GatsbyImageSharpFluid_tracedSVG
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
`;
